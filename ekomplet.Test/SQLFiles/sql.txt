/****** Object:  StoredProcedure [dbo].[remove_supervisor_from_installer]    Script Date: 26-04-2023 13:42:20 ******/

CREATE PROCEDURE [dbo].[remove_supervisor_from_installer]
    @installer_id VARCHAR(36),
    @supervisor_id VARCHAR(36)
AS
BEGIN
    -- Check if the installer and supervisor have a relation in the installer_supervisor table
    IF NOT EXISTS (SELECT * FROM installer_supervisor WHERE installer_id = @installer_id AND supervisor_id = @supervisor_id)
    BEGIN
        RAISERROR ('There is no relation between the installer and supervisor in the installer_supervisor table', 16, 1)
        RETURN;
    END

    -- Check if the supervisor is also in the installers table under supervisor_id
    IF EXISTS (SELECT * FROM installers WHERE supervisor_id = @supervisor_id and installer_id = @installer_id)
    BEGIN
        DECLARE @new_supervisor_id VARCHAR(36)
        
        -- Get a new supervisor for the installer from the installer_supervisor table
        SELECT TOP 1 @new_supervisor_id = supervisor_id 
        FROM installer_supervisor 
        WHERE installer_id = @installer_id AND supervisor_id != @supervisor_id

        -- Check if there is a new supervisor available
        IF @new_supervisor_id IS NULL
        BEGIN
            RAISERROR ('There is no other supervisor to assign to the installer', 16, 1)
            RETURN;
        END		

        -- Update the supervisor_id for the installer in the installers table
        UPDATE installers 
        SET supervisor_id = @new_supervisor_id 
        WHERE installer_id = @installer_id
    END

    -- Delete the relation from the installer_supervisor table
    DELETE FROM installer_supervisor 
    WHERE installer_id = @installer_id AND supervisor_id = @supervisor_id
END
GO
